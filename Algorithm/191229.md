# 알고리즘 - 연구소
 - 1. 실제 bfs와 dfs를 섞어서 문제를 풀어야 했던 좋은 문제
 - 2. dfs를 두번 적용하는데 prev() 단계를 둘 다 적용할 때 실수가 없어야 한다는 점.
 - 3. dfs (nC3) → bfs (infection) 을 3중 for문으로 풀어봐 보는 연습도 해야 할 듯.

CODE
```
#define _CRT_SECURE_NO_WARNINGS
#define MAX 9
#include<iostream>
#include<vector>
#include<queue>
#include<cstring>

using namespace std;
int map[MAX][MAX] = { 0, };
int checked[MAX][MAX] = { 0, };
int dfs_check[90] = { 0, };
struct info {
	int x;
	int y;
};

int dx[4] = { -1, 0, 1, 0 };
int dy[4] = { 0, 1, 0, -1 };
int n, m, depth = 0, cnt = 0, ans = 0, block = 0;
vector<info> v;
vector<info> b; // 0인것만 모아둔 벡터
vector<info> dfs_list;
info bin;

void bfs(info now);
void dfs(int idx, int depth);

int main() {
	
	info srt;
	cin >> n >> m;
	
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &map[i][j]);
			srt.x = i; srt.y = j;
			if (map[i][j] == 2) {
				v.push_back(srt);
				//checked[i][j] = 1;
			}
			if (map[i][j] == 1) {
				block++;
			}
			if (map[i][j] == 0) {
				b.push_back(srt);
			}
		}
	}
	//memcpy(checked2, checked, sizeof(checked));
	// b의 벡터가 뚜루루 있는 상황에서 3개를 골라야 함.
	for (int i = 0; i < b.size(); i++) {
		dfs(i, depth);
		srt = dfs_list.back();
		map[srt.x][srt.y] = 0;
		memset(dfs_check, 0, sizeof(dfs_check));
		dfs_list.clear();
	}

	printf("%d\n", ans);
	
	return 0;
}


void dfs(int idx, int depth) {
	info bin;
	dfs_check[idx] = 1;
	depth++;
	bin.x = b[idx].x; bin.y = b[idx].y;
	dfs_list.push_back(bin);

	if (depth == 3) {
		// map에 idx 표시에 해당하는 값을 넣어야지.
		for (int i = 0; i < dfs_list.size(); i++) {
			map[dfs_list[i].x][dfs_list[i].y] = 1;
		}

		for (int i = 0; i < v.size(); i++) {
			bfs(v[i]);
		}
		// 체킹 후 0의 개수를 세야지
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (checked[i][j] == 0) {
					cnt++;
				}
			}
		}
		cnt = cnt - (block + 3 + v.size());
		ans = max(ans, cnt);
		//memcpy(checked, checked2, sizeof(checked));
		cnt = 0;

		return;
	}
	else {
		for (int i = idx; i < b.size(); i++) {
			if (dfs_check[i] == 0) {
				dfs(i, depth);
				bin = dfs_list.back();
				dfs_check[i] = 0;
				map[ bin.x ][bin.y ] = 0;
				dfs_list.pop_back();
				memset(checked, 0, sizeof(checked));
			}
		}
	}
}
void bfs(info now) {
	queue<info> q;
	info temp;
	int x, y, nx, ny;
	q.push(now);

	while (!q.empty()) {
		temp = q.front(); q.pop();
		x = temp.x; y = temp.y;

		for (int i = 0; i < 4; i++) {
			nx = x + dx[i];
			ny = y + dy[i];

			if (0 > nx || nx >= n || 0 > ny || ny >= m) continue;
			
			// checked와 dist를 동시에 나타내는법 없나
			if (map[nx][ny] == 0 && checked[nx][ny] == 0) {
				temp.x = nx; temp.y = ny;
				
				q.push(temp);
				checked[nx][ny] = checked[x][y] + 1;
			}
		}
	}

}
```

 ```
 int dx[4] = { 0, 1, 0, -1 };
	int dy[4] = { 1, 0, -1, 0 };

	info tmap[2][105];
	int check[2][105] = { 0, }; // 이러면 전체 초기화되냐?
	int d = 0;
	int sx = 0, sy = 0;
	memset(check, 0, sizeof(check));

	while (1) {	
		int nx = sx + dx[d];
		int ny = sy + dy[d];
		if (nx < 0 || nx >= 2 || ny < 0 || ny >= n) {
			d = (d + 1) % 4;
			continue;
		}
		if (check[nx][ny] == 1) break;
		tmap[nx][ny] = map[sx][sy];
		check[nx][ny] = 1;
		sx = nx, sy = ny;
	}

	memcpy(map, tmap, sizeof(tmap));
 ```
